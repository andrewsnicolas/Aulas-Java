package Aulas.Aula10.src.main.java;
import java.io.File;
import java.util.Scanner;
public class aula10 {
    public static void main(String args[]){
        Scanner reader = new Scanner(System.in);
        System.out.println("Digite o nome do arquivo que deseja criar: ");
        String nome = reader.next();
        nome+= ".txt";
        System.out.println(nome);
        File arquivo = new File(nome);
        try{
            if(arquivo.createNewFile()){
                System.out.println("Arquivo foi criado");
            } else{
                System.out.println("Arquivo já existe");
            }
        } catch(IOException e){
            System.out.println("Houve um erro no processo de criar um arquivo.");
        }
    }
}
